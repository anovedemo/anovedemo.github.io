query FilterTask($arn: String = "anove:company:draftsman") {
  Task {
    filter(key_begins_with: $arn) {
      resultset {
        key
        dueDate
        notes
        completeDate
        arn
        owner
        evidence
        category
        assessedMaturityLevel
        assessedMaturityDescription
        employee {
          username
          user {
            lastName
            firstName
          }
        }
      }
    }
  }
}
