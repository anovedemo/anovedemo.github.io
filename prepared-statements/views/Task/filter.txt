query FilterTask($key_begins_with: String = "") {
  Task {
    filter(key_begins_with: $key_begins_with) {
       resultset { 
        key
        dueDate
        notes
        completeDate
        arn
        owner
        evidence
        category
        assessedMaturityLevel
        assessedMaturityDescription
      }
    }
  }
}
